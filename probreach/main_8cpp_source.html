32 void parse_pdrh(string filename)
33 {
34  CLOG_IF(global_config.verbose, INFO, "parser") << "Model file: " << filename;
35  FILE *pdrhfile = fopen(filename.c_str(), "r");

38  CLOG(ERROR, "parser") << "Couldn't open " << filename;
39  exit(EXIT_FAILURE);
40  }
41  std::stringstream s, pdrhnameprep;
42  pdrhnameprep << filename << ".preprocessed";
43  s << "cpp -w -P " << filename << " > " << pdrhnameprep.str().c_str();
44  int res = system(s.str().c_str());
45  // cheking the result of system call
46  if(res != 0)
47  {
48  CLOG(ERROR, "parser") << "Problem occured while preprocessing " << filename;
49  exit(EXIT_FAILURE);
50  }
51  // parsing the preprocessed file
52  FILE *pdrhfileprep = fopen(pdrhnameprep.str().c_str(), "r");
53  // make sure it's valid:
54  if (!pdrhfileprep)
55  {
56  CLOG(ERROR, "parser") << "Couldn't open " << pdrhnameprep.str();
57  exit(EXIT_FAILURE);
58  }
59  // set lex to read from it instead of defaulting to STDIN:
60  yyin = pdrhfileprep;
61  // parse through the input until there is no more:
62  do

65  }
66  while (!feof(yyin));
