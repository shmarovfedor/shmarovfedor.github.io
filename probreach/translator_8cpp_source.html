352 string translator::Translator::translate_jump_guard(pdrh::node *guard, int mode_id)
353 {
354  stringstream s, value;
355  if (pdrh::var_map.count(guard->value)){
356  value << slStateBase << mode_id << "." << guard->value;
357  } else if (guard->value == "="){
358  value << "==";
359  } else if (guard->value == "and"){
360  if (guard->operands.size() == 1){
361  translate_jump_guard(guard->operands.front(), mode_id);
362  } else {
363  value << " & ";
364  }
365  } else if (guard->value == "or"){
366  value << " | ";
367  }
368  else {
369  value << guard->value;
370  }
371  // checking whether it is an operation node
372  if(guard->operands.size() > 1)
373 
374  {
375  s << "(";
376  for(int i = 0; i < guard->operands.size() - 1; i++)
377  {
378  s << translate_jump_guard(guard->operands.at(i), mode_id);
379  s << value.str();
380 
381  }
382  s << translate_jump_guard(guard->operands.at(guard->operands.size() - 1), mode_id) << ")";
383  }
384  else if(guard->operands.size() == 1)
385  {
386  if(value.str() == "-")
387  {
388  s << "(" << value.str() << translate_jump_guard(guard->operands.front(), mode_id) << ")";
389  }
390  else
391  {
392  s << value.str() << "(" << translate_jump_guard(guard->operands.front(), mode_id) << ")";
393  }
394  }
395  else
396  {
397  s << value.str();
398  }
399  return s.str();
400 }

688 string translator::get_initial_value(string variable_name){
689  //TODO: add the other random distributions
690  bool located = false;
691  string lower_bound, upper_bound;
692  for(pdrh::state s : pdrh::init){
693  for(unsigned int i = 0; i < s.prop->operands.size(); i++){
694  pdrh::node* node = s.prop->operands.at(i);
695  if (node->operands.at(0)->value == variable_name){
696  located = true;
697  if (node->value == "="){
698  return translator::resolve_variable_initial_condition(node->operands.at(1));
699  } else {
700  if (node->value == ">="){
701  lower_bound = translator::resolve_variable_initial_condition(node->operands.at(1));
702 // lower_bound = node->operands.at(1)->value;
703  } else if (node->value == "<="){
704  upper_bound = translator::resolve_variable_initial_condition(node->operands.at(1));
705 // upper_bound = node->operands.at(1)->value;
706  }
707  }
708  }
709  }
710  }
711 
712  if (located) {
713  return lower_bound + "+(" + upper_bound + "-" + lower_bound + ").*rand(1,1)";
714  // If we didn't find the variable above...
715  } else {
716  CLOG(WARNING, "translator" ) << "Initial condition for " << variable_name
717  << " not defined - using lower bound of its domain as initial condition.";
718  // Look in the var-map for its interval and use the lower bound as default value
719  auto iter = pdrh::var_map.find(variable_name);
720  if (iter != pdrh::var_map.end()){
721  return iter->second.first->value;
722  }
723  // as a further fail-safe, return 0
724  else {
725  CLOG(WARNING, "translator" ) << "Variable " << variable_name << "not found in variable map, setting initial value to 0";
726  return "0";
727  }
728  }
